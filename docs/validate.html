library(readxl)

# 2. Load the data, validation rules, and code list ###########################
## 2.1 Load the data ----------------------------------------------------------
dog_walks <- readxl::read_excel(
  path = "data/dog_walks.xlsx",
  sheet = "dog_walks")

## 2.2 Load the validation rules -----------------------------------------------
rules <- validate::validator(.data = read.csv(file = "data/dog_walks_rules.csv"))

## 2.3 View the validation rules -----------------------------------------------
View(as.data.frame(rules))

## 2.4 Load the code list ------------------------------------------------------
code_list <- yaml::yaml.load_file(input = "data/dog_walks_code_list.yml")

## 2.5 View the code list ------------------------------------------------------

# 3. Confront the data with the validation rules ##############################
validation <- validate::confront(
  dat = dog_walks
  , x = rules
  , key = "walk_id")
